#include <cstring>
#include <iostream>
#include <cstdio>
using namespace std;
int N;
int V1[1000];
int V2[1000];
int main() 
{
   freopen("agent.in","r",stdin);
   freopen("agent.out","w",stdout);
   V1[++V1[0]]=1;
   cin>>N;
   for(int i=2;i<=N;i++)
   {
      V2[0]=0;
      int nr=0,fr=0;
      for(int j=1;j<=V1[0];j++)
      {
         if(V1[j]!=nr)
         {
            if(nr)
            {
               V2[++V2[0]]=fr;
               V2[++V2[0]]=nr;
            }
            nr=V1[j];
            fr=1;
         }
         else fr++;
      }
      V2[++V2[0]]=fr;
      V2[++V2[0]]=nr;
      memcpy(V1,V2,sizeof(V2));
   }
   for(int i=1;i<=V1[0];i++)cout<<V1[i];
}
